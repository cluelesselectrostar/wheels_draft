import 'package:flutter/material.dart';
import 'dart:async' show Future;
import 'dart:convert';
import 'package:http/http.dart' as http;

Future<RouteStops> fetchRouteStops(String route) async {
  final response =
      await http.get("http://search.kmb.hk/KMBWebSite/Function/FunctionRequest.ashx?action=getstops&route="+ route +"&bound=1&serviceType=1");

  if (response.statusCode == 200) {
    return RouteStops.fromJson(json.decode(response.body));
  } else {
    throw Exception('Failed to load information');
  }
}

class RouteStops{
  Data data;

  RouteStops({this.data});

  factory RouteStops.fromJson(Map<String, dynamic> json) {
    return new RouteStops(
      data: json["data"] as Data,
    );
  }
}

class Data {
  BasicInfo basicInfo;

  Data({this.basicInfo});

  factory Data.fromJson(Map<String, dynamic> json) {
    return new Data(
      basicInfo: json["basicInfo"] as BasicInfo,
    );
  }
}

class BasicInfo {
  String destEName;
  String destCName;
  String oriCName;
  String oriEName;

  BasicInfo ({this.destCName, this.destEName, this.oriCName, this.oriEName});

  factory BasicInfo.fromJson(Map<String,dynamic> json) {
    return new BasicInfo(
      destCName: json["DestCName"] as String,
      destEName: json["DestEName"] as String,
      oriCName: json["OriCName"] as String,
      oriEName: json["OriEName"] as String,
    );
  }
}

  String _setImage(String operator, String lantauTag) {
    if (operator == "kmb" && lantauTag!= null) {
      return 'images/lwb.png';
    } else if (operator == "kmb") {
      return 'images/kmb.png';
    } else if (operator == "nwfb") {
      return 'images/nwfb.jpg';
    } else if (operator == "ctb"){
      return 'images/ctb.png';
    } else {
      return 'images/joint.png';
    }
  }

  Icon _setTagIcon(String tag, String lantauTag) {
    if (lantauTag == "Airport"){
      return Icon(Icons.flight_takeoff);
    } else if (tag == "Peak") {
      return Icon(Icons.directions_run);
    } else if (tag == "Special") {
      return Icon(Icons.priority_high);
    } else if (tag == "Racecourse") {
      return Icon(Icons.monetization_on);
    } else if (tag == "Night") {
      return  Icon(Icons.brightness_2);
    } else if (tag == "Border") {
      return Icon(Icons.leak_remove);
    } else if (tag == "HST") {
      return Icon(Icons.train);
    }
    return null;
  }
  
  String _orgToDest() {

    @override
    void initState() {
      super.initState();
      futureRouteStops = fetchRouteStops(current_route);
    }
    //if (operator == "kmb") {
      FutureBuilder<RouteStops>(
            future: futureRouteStops,
            builder: (context, snapshot) {
  
              if (snapshot.hasData) {
                  String oriCName = snapshot.data.data.basicInfo.oriCName;
                  String destCName = snapshot.data.data.basicInfo.destCName;
                  return Text(oriCName + "-" + destCName);
              } else if (snapshot.hasError) {
                return Text("${snapshot.error}");
              }

              return CircularProgressIndicator(
                value: 18,
              );
            }
        );

    //}
  }

  Widget build(BuildContext context) {
    return ListView.builder(
      itemCount: _routes.length,
      itemBuilder: (context, index) {
        current_route = _routes[index]["route_number"];
        return Card (
            child: ExpansionTile(
              leading: Image(
                image: new AssetImage(_setImage(_routes[index]["operator"], _routes[index]["lantau_tag"])),
                width: 40,
              ),
              title: Text(
                _routes[index]["route_number"] //+ _orgToDest(),

                Align(

                  )
                ),
              ],
              trailing: _setTagIcon(  _routes[index]["tag"], _routes[index]["lantau_tag"]),
